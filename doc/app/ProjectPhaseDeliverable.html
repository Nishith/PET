def complexity_str
  if !self.complexity.blank? &&
      self.complexity >= 0 &&
      self.complexity < LifecyclePhaseDeliverable::COMPLEXITY.size then
    LifecyclePhaseDeliverable::COMPLEXITY[self.complexity]
  else
    ""
  end
end
